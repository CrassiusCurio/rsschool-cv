    function fib_rabbits(n, b) {
        let mature = 1;
        let offspring = 0;
        let immature = 0;
        for (i = 1; i < n; i++) {
            immature = offspring;
            offspring = mature * b;
            mature += immature;
        }
        return n == 0 ? n : mature;
    }
